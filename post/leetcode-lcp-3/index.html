LCP 3. 机器人大冒险

这道题目考察对循环节的理解。

#include <vector>
#include <iostream>
#include <cmath>

using namespace std;

class Solution {
public:
  int deltax, deltay;
  bool robot(string command, vector<vector<int>>& obstacles, int x, int y) {
    deltax = 0, deltay = 0;
    for (auto i: command)
      if (i == 'U') deltay++;
      else deltax++;
    for (auto i: obstacles) {
      if (i[0] > x || i[1] > y) continue;
      if (check(command, i[0], i[1], i[0], i[1])) return false;
    }
    return check(command, x, y, x, y);
  }
  bool check(string cmd, int x, int y, int X, int Y) {
    int N, startx, starty, N1;
    if (cmd[0] == 'U') {
      N = ceil(1. * y / deltay);
      N1 = ceil(1. * x / deltax);
      if (x % deltax == 0) {
      if (!(N == N1 || N == N1 + 1)) return false;
      } else
      if (N != N1) return false;

      startx = (N - 1) * deltax;
      starty = (N - 1) * deltay;
    } else {
      N = ceil(1. * x / deltax);
      N1 = ceil(1. * y / deltay);
      if (y % deltay == 0) {
      if (!(N == N1 || N == N1 + 1)) return false;
      } else
      if (N != N1) return false;

      startx = (N - 1) * deltax;
      starty = (N - 1) * deltay;
    }
    if (startx == X && starty == Y) return true;
    for (auto i: cmd) {
      if (i == 'U') starty++;
      else startx++;
      if (startx == X && starty == Y) return true;
    }
    return false;
  }
};

int main(void) {
  Solution a;
  using VI = vector<int>;
  auto v = vector<VI>{};
  cout << "re: " << a.robot("URR", v, 3, 2) << endl;
  v = vector<VI>{{2, 2}};
  cout << "re: " << a.robot("URR", v, 3, 2) << endl;

  v = vector<VI>{{4, 2}};
  cout << "re: " << a.robot("URR", v, 3, 2) << endl;

  v = vector<VI>{{7, 7}, {0, 5}, {2, 7}, {8, 6}, {8, 7}, {6, 5}, {4, 4}, {0, 3}, {3, 6}};
  cout << "re: " << a.robot("URRURRR", v, 4915, 1966) << endl;

  return 0;
}
